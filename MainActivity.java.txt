package com.example.pizzarend;

import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.provider.MediaStore;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.RadioButton;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {

    private EditText vnev;
    private EditText knev;
    private Button button;
    private RadioButton elso;
    private RadioButton masodik;
    private RadioButton harmadik;
    private TextView textView;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        findViews();

        button.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                Intent i = new Intent(MainActivity.this, rendeles.class);

                i.putExtra("vezeteknev",vnev.getText().toString());
                i.putExtra("keresztnev", knev.getText().toString());




                startActivity(i);

                if(elso.isChecked()){
                    textView.setText("24 cm");
                    i.putExtra("meret", textView.getText().toString());
                }
                else if(masodik.isChecked()){
                    textView.setText("28 cm");
                    i.putExtra("meret", textView.getText().toString());

                }
                else if(harmadik.isChecked()) {
                    textView.setText("40 cm");
                    i.putExtra("meret", textView.getText().toString());
                }
            }
        });

    }

    private void findViews() {
        vnev = findViewById(R.id.vnev);
        knev = findViewById(R.id. knev);
        button = findViewById(R.id.button);
        elso = findViewById(R.id.elso);
        masodik = findViewById(R.id.masodik);
        harmadik = findViewById(R.id. harmadik);
        textView = findViewById(R.id. textview2);
    }
}
